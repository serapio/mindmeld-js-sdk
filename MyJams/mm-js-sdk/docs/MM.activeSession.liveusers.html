Namespace: liveusers

MM.activeSession. liveusers

MM.activeSession.liveusers represents the LiveUsers collection in the MindMeld API. The list of User objects who are currently using the Session

Methods

delete(liveuserid, onSuccess, onFail)

Deletes a user from the list of active users for the active session

Name Type Description
liveuserid string

id of the user to remove from active user list
onSuccess APISuccessCallback optional

callback for when removing user from active users list was successful
onFail APIErrorCallback optional

callback for when removing user from active users list failed
Example
         function removeLiveUser () {
            MM.activeSession.liveusers.delete('365', onLiveUserRemoved);
         }
         function onLiveUserRemoved () {
            // live user removed
         }

get(params, onSuccess, onFail)

Get the list of users that are currently active users of the specified session. A request with a user token can get the liveusers list only if the associated user is permitted to access the session object itself. A request with an admin token can get the liveusers list for any session associated with your application.

Name Type Description
params QueryParameters optional

query parameters when fetching the live user list
onSuccess APISuccessCallback optional

callback for when getting live user list was successful
onFail APIErrorCallback optional

callback for when getting live user list failed
Example
         function getLiveUsers () {
            MM.activeSession.liveusers.get(null, onGetLiveUsers);
         }
         function onGetLiveUsers (response) {
            var liveUsers = response.data;
            console.log(liveUsers);
         }

json(){Array.<Object>}

Helper function returns the JSON data for the live users list

Example
         function getLiveUsers () {
            MM.activeSession.liveusers.get(null, onGetLiveUsers);
         }
         function onGetLiveUsers () {
            var liveUsers = MM.activeSession.liveusers.json();
            // MM.activeSession.liveusers.json() returns a JSON object
            // containing data received from MM.activeSession.liveusers.get()
         }

onUpdate(updateHandler, onSuccess, onError, getQueryParams)

Sets the activeSession.liveusers' onUpdate handler. If no handler is passed in, onUpdate unsubscribes from push events

Name Type Description
updateHandler APISuccessCallback optional

callback for when the activeSession's live users list updates
onSuccess function optional

callback for when subscription to onUpdate event succeeds
onError function optional

callback for when subscription to onUpdate event fails
getQueryParams QueryParamGetter optional

custom function used to determine QueryParameters used to in get() request when collection updates
Examples

Setting the onUpdate handler, adding a live user, and obtaining the latest live user list 

         function liveUsersOnUpdateExample () {
            // set the onUpdate handler for the liveusers list
            MM.activeSession.liveusers.onUpdate(
                onLiveUsersUpdate,
                onSubscribedToLiveUsersUpdates,
                onSubscribeToLiveUserUpdatesError,
                getLiveUserListParams
            );
         }
         function onSubscribedToLiveUsersUpdates () {
            // successfully subscribed to updates to the session's liveusers list
            console.log('subscribed');
            // now, add a live user
            addLiveUser();
         }
         function onSubscribeToLiveUserUpdatesError () {
            console.log('error subscribing to live user list updates');
         }
         function getLiveUserListParams () {
            // When the live user list updates, only fetch 5 objects
            return {
                limit: 5
            };
         }
         function onLiveUsersUpdate () {
            // there was an update to the liveusers list
            var liveusers = MM.activeSession.liveusers.json();
            console.log(liveusers);
            // liveusers contains the latest list of liveusers
         }
         function addLiveUser () {
            var liveUserData = {
                userid: '365'
            };
            MM.activeSession.liveusers.post(liveUserData);
         }

Deregistering the onUpdate handler 

         function deregisterLiveUsersOnUpdate () {
            MM.activeSession.liveusers.onUpdate(null);
         }

post(newLiveUserData, onSuccess, onFail)

Adds a new user to the list of active users for the active session

Name Type Description
newLiveUserData Object

object specifying userid of user to be added to live user list

Name Type Description
userid string

The MindMeld userid for the user to add to the liveusers list for the session
onSuccess APISuccessCallback optional

callback for when adding live user was successful
onFail APIErrorCallback optional

callback for when adding live user failed
Example
         function addLiveUser () {
            var liveUserData = {
                userid: '365'
            };
            MM.activeSession.liveusers.post(liveUserData, onLiveUserAdded);
         }
         function onLiveUserAdded (response) {
            // New live user added
         }
